função acharSoma(array:vetor, n:int, soma:int):
    se soma = 0:
        retorna 1
    senão se n <= 0:
        retorna 0
    senão se tabela[n,soma] != -1:
        retorna tabela[n,soma]
    senão se array[n] > soma:
        tabela[n,soma] = acharSoma(array, n - 1, soma)
        retorna tabela[n,soma]
    senão:
        tabela[n,soma] = acharSoma(array, n - 1, soma)
        retorna tabela[n,soma] or acharSoma(array, n - 1, soma - array[n - 1])

se acharSoma(array, n, soma) == 1:
    imprime(f"Existe um subconjunto para a soma {soma}")
senão:
