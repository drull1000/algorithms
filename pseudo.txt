int n = input("Qual a quantidade de vezes que o padrão se repete? ")
str message = input("Qual a sua mensagem? ")
str code = input("Qual o código? ")

se 0 > n > 1000:
    imprime("Padrão grande demais. O limite é 1000")
    sair()
se 0 > tamanho(message) > 1000:
    imprime("Mensagem grande demais. O limite é 1000")
    sair()
se 0 > tamanho(code) > 100:
    imprime("Código grande demais. O limite é 100")
    sair()

função generatearray(message: str, code: str): array
    int array array = [-1 para i até (tamanho(message))]
    para i=1 até (tamanho(code)):
        para j=1 até (tamanho(message)):
            se code[i] = message[j]:
                array[j] = i
    remover(array, -1)
    retornar array

array = generatearray(message: str, code: str)

função verifyMessage(n: int, array: array): int
    int i = 0
    enquanto i < tamanho(array)-1:
        se array[i+1] <= array[i]:
            array.remover(i+1)
            n--
            i = 0
        i++
    se n = 0:
        retornar 1
    senão:
        retornar 0

se verifyMessage(n-1,array) = true
    imprime("Y")
senão
    imprime("N")
